Criminal Network

Home / Demo / Criminal Network
Criminal Network

Renderer: SVG/WebGL

Module: vizart-network

Data: Federal criminal records are transformed and stored in Neo4J. Visualizations are created on Neo4J Cypher queries. 

const bossQuery = "MATCH (a:Person{personNo:\"2005282\"})-[r1:CALLS]-(b:Person)-[r2:PARTICIPATES]-(c:Case)" +
	" SET a.weight=7, b.weight=3, c.weight=7, a.role='boss', b.role='member'," +
	" c.role='case', r1.strength=10, r2.strength=10" +
	" RETURN a, b, c, r1, r2  limit 30";


const query2 = "MATCH (a:Person)-[r1:CALLS]-(b:Person)-[r2:PARTICIPATES]-(c:Case) "
	+ " WHERE NOT ((a)-[:PARTICIPATES]-(:Case)) "
	+ " SET a.weight=7, b.weight=3, c.weight=7, a.role='boss', b.role='member', c.role='case', r1.strength=10, r2.strength=10  "
	+ " RETURN a, b, c, r1, r2, count(b) AS cnt ORDER BY cnt DESC limit 300 ";

const query3 = "MATCH (b:Person)-[r2:PARTICIPATES]-(c:Case) " +
	" SET b.weight=3, c.weight=7, b.role='member'," +
	" c.role='case', r2.strength=10" +
	" RETURN b, c, r2 limit 1000";
